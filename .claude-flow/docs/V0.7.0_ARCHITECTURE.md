{% macro span(name, kind, attrs={}) %}

kind = "{{ kind }}"
{% if attrs %}
attrs.all = { {% for k, v in attrs %}"{{ k }}" = "{{ v }}"{{ ", " if not loop.last }}{% endfor %} }
{% endif %}

{{ self::span(service ~ ".exec", "internal") }}

  ["{{ service }}.start", "{{ service }}.exec"],

{% for parent, child in pairs %}
